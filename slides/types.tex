\subsection{Types}
    \begin{frame}
        \frametitle{Types}

        Everything in elm has a type.
        \newline
        
        \begin{example}
            varInteger : Int\\
            varInteger = 1234
        \end{example}
        \begin{example}
            varInteger = "This is a string, not an integer!"\\
            
            \quad\textit{ -- the compiler infers varInteger is a String}
        \end{example}
        In Elm all variables are immutable!
        
    \end{frame}

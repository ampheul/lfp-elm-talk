\subsection{Records}
    \begin{frame}

        \frametitle{Records}
        Elm has records.
    
        \begin{codeexample}
            record : \{name : String, age : Int\}\\
            record = \\
            $\quad$\{ name = "sam"\\
            $\quad$ , age = 99\\
            $\quad$\}
        \end{codeexample}

        Records are used a lot in elm in combination with type aliases.

    \end{frame}
